1、事件对象兼容---------低版本ie不识别事件参数，只识别event

2、鼠标事件对象兼容--------偏移量兼容，火狐不识别offsetX/offsetY，识别layerX/layerY
    【解决】：var dis = offsetX || layerX;
3、鼠标事件对象兼容--------鼠标键值不同，低版本的ie的键值不同
正常：    0/1/2 左键/点击滚轮/右键   低版本ie：   1/4/2左键/点击滚轮/右键
    【解决】：
function fn(evt){
    if( evt ){    //直接写事件参数，用事件参数判断，如果是高版本识别传参，低版本不识别传参但识别event
    
        return evt.button
    }else if(event){
        switch( event.button ){
            case 1 : return 0;
            case 4 : return 1;
            case 2 : return 2;
        }
       }
    }

4、键盘事件兼容--------获取键盘按键值    keyCode     但火狐不识别，火狐识别which
    【解决】：var code = e.keyCode || e.which    【注】：e.which是火狐识别的获取键盘按键值
5、其他事件兼容--------阻止事件冒泡兼容    
    【解决】：
e.stopPropagation ? e.stopPropagation() : e.cancelBubble = true;    【注】:函数名等于函数本身（js解析器知识 ），布尔环境下，有函数为真，为空为假            propagation：传播        cancle：取消
e.stopPropagation();//高版本的阻止冒泡，stoppropagation是事件对象属性
e.cancelBubble = true;//ie的阻止冒泡

e.preventDefault ? e.preventDefault() : e.returnValue = false;
7、其他事件兼容-------获取页面滚走距离        因为ie的最上层是body
    【解决】：var scroll = document.documentElement.scrollTop  || document.body.scrollTop
8、其他事件兼容-------getElementsByClassName()    ie不识别
    【解决】：封装函数
兼容document.getElementsByClassName()
function byClassName(cname){
       //判断高版本浏览器,如果是高版本浏览器识别并返回，低版本不能识别则该函数为null，布尔环境下null为空
       if( document.getElementsByClassName ){
           return document.getElementsByClassName(cname);
       }
       //低版本浏览器   定义<h1></h1>一个空数组   遍历页面上所有的对象  判断某个对象的className==cname 
       //满足这个条件 ，就将该元素存入到数组中 
       var arr = [];
       var all = document.getElementsByTagName("*");//找页面上所有的元素
       for( var i = 0; i <  all.length; i++ ){
           if( all[i].className == cname ){
               arr.push(all[i])
           }
       }
       return arr;
    }
