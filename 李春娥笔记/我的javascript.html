兼容document.getElementsByClassName()
    function byClassName(cname){
       //判断高版本浏览器,如果是高版本浏览器识别并返回，低版本不能识别则该函数为null，布尔环境下null为空
       if( document.getElementsByClassName ){
           return document.getElementsByClassName(cname);
       }
       //低版本浏览器   定义<h1></h1>一个空数组   遍历页面上所有的对象  判断某个对象的className==cname 
       //满足这个条件 ，就将该元素存入到数组中 
       var arr = [];
       var all = document.getElementsByTagName("*");//找页面上所有的元素
       for( var i = 0; i <  all.length; i++ ){
           if( all[i].className == cname ){
               arr.push(all[i])
           }
       }
       return arr;
    }

是么是原生js
原生js是指未经过封装的JS代码,而jquery就是一种框架

原生态js是指遵循ECMAscript标准的javascript，不同于微软的jscript也不依赖于任何框架，依托于浏览器标准引擎的脚本语言，jquery是在原生态的js上集成的框架资源，使用jquery并不代表代码就不是你写的，区别在于使用jquery只是使用了它的语法，更利于兼容以及实现，jquery已经将常规的js兼容问题解决，所以使用起来出现不兼容的情况会更少，更利于开发，但同时，要使用jquery等框架也需要去学习相关的语法，拥有js基础的同学学习jquery时会发现很轻松，在使用时也会发现很方便。

获取随机字符及任意长度的验证码
function getRandCh(){//获取一个任意字符大小写字母52个，十个数字
    var num = Math.floor( Math.random() * 62) ;//大小写字母52个，十个数字
    if( num < 10 ){
        return num;//0-10
    }
    if( num < 36 ){
        return String.fromCharCode( num+55 );//A-Z  10+55=65=A   35+55=90=Z
    }else{
        return String.fromCharCode( num+61 );//a-z  36+61=97=a   61+61=122=z
    }
}

function getRandStr(l){//获取任意长度的验证码，包含大小写
    var str = "";
    while(str.length<l){
        str += getRandCh();
    }
    return str;
}
