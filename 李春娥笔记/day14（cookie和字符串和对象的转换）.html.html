什么是cookie ：   会话跟踪技术

会话跟踪技术 ： 浏览器（客户端） 和 服务器 之间在进行多次请求数据的时实现数据共享的过程 就称为  会话跟踪技术

5、生存期     

生存期  就是 cookie数据在浏览器上保存的时间     生存期就是cookie的一个参数
document.cookie = "键=值;expires=标准时间格式"
关于生存期问题：
如果cookie没有生存期，cookie数据会在会话结束后自动消失
如果设置cookie生存期，cookie数据会在cookie生存期设置的时间后消失
如果删除cookie，可以将键的值设置为""  或 将生存期设置为  -1
假删除：手动改动系统时间  

关于cookie几点注意：
1、cookie存储数据不安全 
2、cookie存储数据不能超过4kb  最多不能超过 50个
3、cookie不能跨文件夹访问

function getCookie(key){
    //console.log(document.cookie)//没有cookie时，该值为空，布尔环境下，空字符串解析为false
    if( document.cookie ){
        var str = document.cookie;
        var arr = str.split("; ")//默认cookie字符串没一个cookie值是以一个分号加一个空格分隔
        for( var i= 0; i < arr.length; i++ ){
            var brr = arr[i].split("=");
            if( brr[0]==key ){
                return brr[1];
            }
        }
        //有cookie数据，但是没有key 得不到cookie，返回一个空字符串
        return "";
    }
    //如果没有cookie
    return "";
}
 
封装一个函数设置cookie值

function setCookie(key,value,day){
    if(day){
    var d = new Date();
    d.setDate(d.getDate() + day);
    document.cookie = key + "=" + value +";expires=" + d;
    }else{
        document.cookie = key + "=" +value;
    }
}

删除cookie

function removeCookie(key){
    setCookie(key,"",-1)
}
