2、数学对象 Math  特点 ： 不需要定义  直接通过Math.方法名 调用方法

Math.floor() 向下取整  4.3 -- 4   -4.3 --- -5
Math.ceil()  向上取整  4.3 -- 5   - 4.3 --- -4
Math.round()  四舍五入  
Math.sqrt(m)  开平方    
Math.pow(m,n)  m的n次方
Math.min( 342,1,45 )  1
Math.min.apply( null, [34,2,45] )  2  获取数组的最小值
Math.random() 随机 获取 [0,1) 随机小数
Math.abs()  取绝对值


    function rand(min,max){//随机数
       return Math.round( Math.random()*(max-min) + min );
    }
    function getColor(){//随机颜色
       var r = rand(0,255);
       var g = rand(0,255);
       var b = rand(0,255);
       var color = "rgb(" + r + "," + g + "," + b + ")"
       return color;
    }

4、日期对象 Date

定义 ： 
    var d = new Date();

获取时间方法 ： 
getFullYear()  年
getMonth() 月份   ( getMonth() + 1 )  月份 0--11        【注】：getMonth()获取的结果是0-11结果加1就是当前月份
getDate()  日期
getDay()   星期  0--6   0  --  星期日 
getHours()   时
getMinutes() 分 
getSeconds()  秒

d.toLocaleString()  转成本地时间格式方法

    var d = new Date();
    console.log(d)//Fri Mar 16 2018 22:47:16 GMT+0800 (中国标准时间)
    console.log(d.getFullYear())//2018
    console.log(d.getMonth())//2 --- 所以获取当月需要 d.getMonth()+1
    console.log(d.getDate())//16 --- 获取天是getDate
    console.log(d.getDay())//5 --- 获取星期几是getDay，返回值都是number类型
    console.log(d.getHours())//22
    console.log(d.getMinutes())//54
    console.log(d.getSeconds())//26
    d.setDate(d.getDate()+10)//cookie经常用，设置天数
    console.log(d.getDate())//26

getTime()

    console.log(end.getTime())//1521212400000
    function diff(start,end){
       return (end.getTime() - start.getTime())/1000;//查多少秒
    }
    console.log(diff(end,d));//203.174
