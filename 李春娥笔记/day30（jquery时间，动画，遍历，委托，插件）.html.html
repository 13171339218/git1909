4、jquery事件

mouseover 和 mouseenter 区别 ：   前者会冒泡 后者不会产生冒泡
事件绑定 ：
    bind()
    用法一 : $().bind("事件",function(){  })
    用法二 : $().bind("事件1 事件2",function(){  })
    用法三 ：$().bind({ 
                事件1 : function(){},
                事件2 : function(){},
                ....
              })
    取消绑定 ：unbind()   $().unbind( "事件" )

     $("#box").bind("mousedown",function(e){
           var e = e || event;
           //console.log($("#box").offset().left)
           var disx = e.pageX - $("#box").offset().left;
           var disy = e.pageY - $("#box").offset().top;
           //console.log(disx+","+disy)
           $(document).bind("mousemove",function(e){
                var e = e || event;
                $("#box").css({left:e.pageX-disx,top:e.pageY-disy})
           }).bind("mouseup",function(){
                //document.onmousemove = null;//没用，jquery添加的事件只能通过jquery删除
                $(document).unbind("mousemove");
           })
           return false;
     })

事件委托 :
    delegate :                                         delegate：委托
    $().delegate("事件源","事件",function(){   })
    取消委托 ：
    undelegate()   $().undelegate()

     $("ul").delegate("li","click",function(){
           //console.log(this)
           //this.style.background = "red";//可以使用
           this.css("background","green")//报错 this.css is not a function 说明用jquery添加的事件中this指向的是dom对象
     })
     $("ul").undelegate();
     
     $("li").on("click",function(){
           console.log(this)
           this.style.background = "red";//可以使用
          //this.css("background","green")//报错 this.css is not a function
     })

事件绑定 + 事件委托 ：
    on :
        实现事件绑定 ：  用法 同  bind() ，也有三种用法，完全同bind

     $("li").on({
          "mouseenter":function(){$(this).css("background","gold")},
          "mouseleave":function(){$(this).css("background","pink")}
     });
     $("li").off()//取消绑定在该元素上的所有元素
 
        实现事件委托 :  $().on("事件","事件源",function(){  })
    off ： 取消事件绑定或委托


     $("ul").on("click","li",function(){//on委托
           $(this).css("background","gold");
     });
     $("ul").off("click");

6、offset方法和position()
offset()获取 相对于文档顶部和左侧的偏移  结果是一个对象  有两个属性  left  top
position()  获取离该元素最近的具有定位的元素的偏移  结果是一个对象  left 和 top属性

width()/height()  获取元素的宽和高 （内容）
innerWidth()/innerHeight()   内容+padding
outerWidth()/outerHeight()   内容+padding+border

     var box1 = $id("box1");
     console.log( $("#box1").width() )//500
     console.log( $("#box1").css("width") )//500px
     console.log( box1.width )//undefined
     console.log( window.getComputedStyle(box1,null)["width"] )//500px
     //获取对象宽度，js中   obj.width只能获取行内的宽度而jquery中  $(obj).width()可以获取到任意位置的宽度，返回值是number
     类型，css中的获取不到，而jquery和window.getComputedStyle(obj,null)[attr],都能获取到，返回值是带单位的字符类型，
     
     console.log( $("#box1").innerWidth() )//520
     console.log( $("#box1").outerWidth() )//522
     
     
     console.log( box1.clientWidth )//520
     console.log( box1.offsetWidth )//522
     jquery中$(obj).innerWidth() 是获取对象内容加padding的宽度值   类似于dom对象的obj.clientWidth()
     jquery中$(obj).outWidth() 是获取对象内容加padding加边框的宽度值   类似于dom对象的obj.offsetWidth()

        .... 
    })

插件定义二（局部） :
    $.fn.extend({
        方法名 : function(){},
        方法名 :  function(){},
        .... 
    })
    调用 ： 选择器.方法名()

参数合并（对象合并）:
console.log( $.extend({a:1,b:2},{a:2,c:3},{b:6}) );//{a:2,b:6,c:3}
