1、严格模式  "use strict"   
    只写在程序的开始（script下的第一句，写在其他位置没有用）----限制整个程序   
    或者
    写在函数的开始（function函数{花括号中的第一句}）---限制函数体内部代码
严格模式常见的要求：
    定义变量必须要有var  
    严格模式下八进制 被禁用
   "use strict"
   var a = 023;//严格模式下八进制 被禁用，八进制是用0开头的
   console.log(a)
    function 严格模式("use strict")必须声明在程序的最顶端    
    arguments与形参不同  非严格模式下如果参数在函数内部改变实参副本arguments中的对应值也会随之改变
    arguments
      说明 ：
            1、只能在函数体内部使用
            2、在函数体中 可以通过arguments.length 获取实参个数  函数.length获取形参的个数
            3、因为arguments代表的是一个集合（对象），取出集合中的某个值 arguments[下标]   下标从0开始 
   function fn(m,b){

3、let 关键字     let  块级作用域 
let 声明变量 
let 不允许重复声明变量      
let 必须在某个{  } 内部使用 
let 在函数体的内部 不能实现 变量提升     暂时性死区 ：  通过let方式声明的变量  变量声明之前该变量不可用  这种现象就称作 暂时性死区 
在函数体内部  不能通过let方式声明和形参相同的变量    【注】：函数形参和函数内部用let创建的值不能同名，否则会包错。
let 应用 ：  获取下标（选项卡）

4、this 指向改变
this含义--当前对象  ： 指事件的调用者 或  方法的调用者  
改变匿名函数的this指向 ： bind(this指向的对象)      bind：捆绑  apply：申请，qingqou     call：呼叫
                       【注】：匿名函数.bind（指向的目标）        bind值能对匿名函数使用
                        apply() 或 call()  改变非匿名函数的this指向     函数名.apply( 函数体内部this指向对象 )
                        fn.call(document.body)//[object HTMLBodyElement]
                        fn.apply(document.body)//[object HTMLBodyElement]
var fn = function(){
    alert(this);
}/*.bind(document.body)*/
fn.apply(document)//通过这种模式定义的也是匿名函数，不过也可以使用apply和call改变this指针中的地址

5、for...of  
for...of  和  for ... in  区别 ： 
for...of 遍历值，不能遍历json对象，通常遍历集合
for ... in 遍历键，只能遍历数组和json对象

    var json = {
           dashaz:15,
           ershaz:17
       }
    for( var i in json ){
       console.log(i)//dashaz,ershaz
    }
    for( var i of json ){
       console.log(i)//报错json is not iterable
    }

    var arr = [1,2,5,7,8];
    for( var i in arr ){
       console.log(i)
    }

11、set集合 和 map集合

Set集合  ：   var set = new Set(); 定义一个set集合  
set集合特点 ： 自动去重   不会进行自动类型转换
实现数组去重 ： Array.from( new Set(arr) )

史上最短的数组去重的方法 ：   [...new Set(arr)]

    var arr = [1,23,5,6,7,7777,7,7,7,71,1,1,1,1,2,3,2,23];
    
    var newArr = Array.from(new Set(arr))//这一句等价于下边的一大段
    console.log(newArr);//(9) [1, 23, 5, 6, 7, 7777, 71, 2, 3]
    
    
    var brr = [];
    var set = new Set(arr);//new Set（）的返回值是去重后的集合
    console.log(set);//Set(9) {1, 23, 5, 6, 7, …}
    set.forEach(function(i){brr.push(i)})//set集合只能用forEach遍历
    console.log(brr)//(9) [1, 23, 5, 6, 7, 7777, 71, 2, 3]
    
    var crr = [...new Set(arr)];
    console.log(crr)//(9) [1, 23, 5, 6, 7, 7777, 71, 2, 3]
forEach(匿名函数)  遍历   没有返回值
   arr.forEach(function(item,index,array){
          item  表示数组中的数据
           index 表示下标
           array 表示遍历的数组
    })

set集合的方法 ： 
add()  向集合中添加一个数
delete（）  删除集合中的某个数
has()  判断集合中是否含有某个值
clear()  清空集合
通过 size  属性 获取集合元素个数 
    var set = new Set();
    set.add(1).add(2);
    set.add(3);
    console.log(set)//Set(3) {1, 2, 3}   object     集合
    console.log(typeof set)
    console.log(set.length)//undefined
    console.log(set.size)//3

Map集合 ---存储的数据以键值对形式存储的 ： 
set(键值对)  向集合中添加一个键值对
delete（键）  根据键删除集合中的某个数
has(键)  根据键判断集合中是否含有某个值
get(键)  根据键取值
clear()  清空集合
var map = new Map();
    map.set("dashaz",15);
    map.set("ershazi",14).set(1,19);
    console.log(map)//Map(3) {"dashaz" => 15, "ershazi" => 14, 1 => 19} object   集合
    console.log(typeof map)
    console.log(map.ershazi)//undefined
    console.log(map.get("ershazi"))//14
    console.log(map[0])//undefined
