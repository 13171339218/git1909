    function sortbuble(arr){//冒泡排序
       for( var i = 0; i < arr.length-1; i++ ){//控制循环轮数
           for( var j = 0; j < arr.length-1-i; j++ ){
               if( arr[j]>arr[j+1] ){
                  [ arr[j],arr[j+1] ] = [ arr[j+1],arr[j] ]//利用解构赋值实现大小数交换
               }
           }
       }
       return arr
    }
    var arr= [1,2,6,4,3]
    console.log(sortbuble(arr))// [1, 2, 3, 4, 6]
    
    
    function changesort(){//选择排序
       for( var i = 0; i < arr.length-1; i++ ){//用第一个数和之后的数比较
           for( var j = i+1; j < arr.length; j++ ){//第一个数之后的数到最后
               if( arr[i]>arr[j] ){
                  [ arr[i],arr[j] ] = [ arr[j],arr[i] ]
               }
           }
       }
       return arr;
    }
    var arr= [1,2,6,4,3]
    console.log(changesort(arr))//[1, 2, 3, 4, 6]

4、数组新增方法

forEach(匿名函数)  遍历   没有返回值
    【注】遍历是从第一个数据遍历到数组中最后一个数据，一遍一遍的走，所以遍历数组的时候会出现数组长度个该数组
   arr.forEach(function(item,index,array){
    【注】匿名函数的第一个值表示数组中的每个空间中的数据，第二个表示空间的下标，第三个表示遍历的数组
       item  表示数组中的数据
           index 表示下标
          array 表示遍历的数组
    })

    var arr= [1,2,6,4,3];
    arr.forEach(function(item,index,array){

    });
