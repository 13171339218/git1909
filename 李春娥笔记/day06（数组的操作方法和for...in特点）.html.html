6、数组的操作方法

length : 获取数组长度  
技巧 ：快速的清空数组 或 快速的改变数组的个数 ：
 var arr = [19,2,3,4,5,6,7];
 arr.length = 0;  清空数组
 arr.length = 1;  保留数组中的第一个数

push()  向数组的尾部添加一个或多个元素   用法 ： arr.push(23,34,12);   返回数组新长度值    【注】：改变原数组
pop()   删除数组尾部的数据  一次删一个   用法 ： arr.pop();  返回 删除的元素     【注】：改变原数组

    var arr = [1,2,3,4];
    console.log(arr);//[1, 2, 3, 4]
    var sb = arr.push("99");
    console.log(sb);//5----返回值是新数组的长度
    console.log(arr);//[1, 2, 3, 4, "99"]
    var num = arr.pop();//弹出最后一个，修改原数组，返回值是弹出的内容
    console.log(arr);// [1, 2, 3, 4]
    console.log(num);//99
    console.log(typeof num);//string  不修改弹出内容的类型

unshift() 向数组的头部添加一个或多个元素  用法 ：arr.unshift(34)      【注】：改变原数组
shift()      删除数组头部的元素        shift：移动         【注】：返回值是删除的元素           【注】：改变原数组

    var arr = [1,2,3,4];
    console.log(arr);//1,2,3,4
    var sb = arr.unshift("chen","15");//一次可以添加多个
    console.log(sb)//6-----返回值是新数组的长度
    console.log(arr)//["chen", "15", 1, 2, 3, 4]
    var char = arr.shift();
    console.log(char)//chen
    console.log(typeof char)//string
    console.log(arr)// ["15", 1, 2, 3, 4]

splice()    splice：拼接           【注】：改变原数组
     一、  arr.splice(start,length)  删除从开始位置 指定长度的数据    返回删除的元素  如果不加length，将start位置后面的数据都删除
     二、  arr.splice( start , length , num参数 );     如果length是0，在start位置处添加num数据     （添加）
                                                      如果length是大于0， 将start位置处的数据先删除，在添加num数据（替换）

     var arr = [1,2,3,4,7,9];
    var brr = arr.splice(2,3)
    console.log(brr);//3,4,7      返回值为删除的内容，放在数组中
    console.log(arr);//1,2,9      第一个数字为开始替换的坐标，第二个为长度，从坐标为2的删除长度为3的内容
    var brr1 = arr.splice(2,0,"99",77,"66")
    console.log(brr1);//[]    返回值为删除的内容,没有删除为空数组
    console.log(arr);//1,2,"99",77,"66",9
    var brr2 = arr.splice(2,1,"char","kang")//1,2,"char","kang",77,"66",7,9  第二个参数，都是替换的内容
    console.log(arr)
    console.log(brr2);//[99]      返回值为删除的内容,长度为1 的数组

上面的五个方法 都会改变原数组 ，且都有返回值  
    push（）在数组尾部添加，添加括号中的内容，可多可少，一次可以添加括号中的全部内容，返回值为新数组长度
    pop（）删除数组尾部最后的数据，一次移除一个数据（最后一个），返回值为删除的数据
    
    shift（）移除数组第一个数据，一次移除一个数据（第一个），返回值为移除的数据
    unshift（）在数组头部添加数据，一次可以添加括号中的全部内容，返回值为新数组的长度

    splice（）拼接，括号内第一个参数是选择开始操作的坐标，第二个为移除的长度，第三个为添加的内容，范围值为删除的内                容，若第二个参数为零，即添加数据，则返回值为空数组
    reverse()颠倒，相反，修改原数组，将原数组的顺序颠倒，返回值为颠倒的新数组，和原数组内容一样

下面这几个方法，  了解方法的返回值  
slice(start,end)  截取从 start 开始处到 end-1 位置处的字符  不会改变原数组   返回截取后的数据   slice：刨切

join()   用法 ：  arr.join("-")  使用 -  将数组中的数据连接起来 转成字符串    数组转成字符串     
        【注】：split()    用法： 字符串方法，将字符串转换成数组    
concat() 用法 ：  arr.concat( brr , crr )   将两个以上的数组合并成一个新数组  不会改变原数组  返回合并后的新数组      concat：合并数组，合并字符串      ps:合并字符串用你？？？加号搞定
reverse()  用法 ： arr.reverse()  数组反转  reverse：颠倒，相反  【注】：会修改原数组，将原数组的顺序颠倒，且返回值也是颠倒顺序的新数组

    var arr = [1,2,3,4];
    var brr = arr.slice(1,3)//返回值是删除的内容
    console.log(brr)//[2, 3],第一个参数是开始操作的坐标，第二个是结束坐标，但不包括结束坐标
    console.log(arr);//[1,2,3,4]   不修改原数组
    var ck = ["chen","kang"]
    var zm = ["zm","15"]
    var newArr = arr.concat(ck,zm)//将两个以上的数组合并为一个新数组，不修改原数组，返回值是新数组
    console.log(newArr)// [1, 2, 3, 4, "chen", "kang", "zm", "15"]
    console.log(ck)//["chen", "kang"]
    console.log(zm)// ["zm", "15"]
    console.log(arr)// [1, 2, 3, 4]
    var rarr =arr.reverse();//修改原数组，返回值为新数组
    console.log(arr)//[4, 3, 2, 1]
    console.log(rarr)//[4, 3, 2, 1]
    console.log(arr.reverse())//[1, 2, 3, 4]


7、栈和堆
栈 ： 当定义一个变量时，计算机就在内存中开辟一块存储空间存放这个变量的值 ，这个空间就叫做 栈 
     栈的特点 ： 后进先出  
     栈中存放的是 基本类型的数据 
    【注】：复合数据类型栈中存放的是地址，数据的地址；地址指向堆中的属性和方法或数据
堆 ： 堆中存放很多个数据  一般堆中存放的是 对象（引用）类型数据

值传递和引用传递的区别 ：
值传递 ： 值之间的改变互不影响
引用传递（地址传递） ：  一改全改，所有指向该位置的数据都会改变  
扩展 ：  在计算机中，有一种特殊的变量，这个变量存放某个数据的地址，这个变量就叫做 指针

8、排序算法 

冒泡排序 ： 相邻的两个数进行比较，大数下沉  小数上浮    从小到大（升序）

    function sortbuble(arr){//冒泡排序
       for( var i = 0; i < arr.length-1; i++ ){//控制循环轮数
           for( var j = 0; j < arr.length-1-i; j++ ){
               if( arr[j]>arr[j+1] ){
                  [ arr[j],arr[j+1] ] = [ arr[j+1],arr[j] ]//利用解构赋值实现大小数交换
               }
           }
       }
       return arr
    }
    var arr= [1,2,6,4,3]
    console.log(sortbuble(arr))// [1, 2, 3, 4, 6]
    
    
    function changesort(){//选择排序
       for( var i = 0; i < arr.length-1; i++ ){//用第一个数和之后的数比较
           for( var j = i+1; j < arr.length; j++ ){//第一个数之后的数到最后
               if( arr[i]>arr[j] ){
                  [ arr[i],arr[j] ] = [ arr[j],arr[i] ]
               }
           }
       }
       return arr;
    }
    var arr= [1,2,6,4,3]
    console.log(changesort(arr))//[1, 2, 3, 4, 6]
